Build a production-grade MVP web application called OriginLock.

OriginLock is a creator verification infrastructure platform for musicians, writers, designers, filmmakers, freelancers, agencies, indie labels, and content creators.

CORE PRODUCT PROMISE:
A user uploads a file and receives a proof-of-creation record that includes:
1. a SHA-256 file fingerprint
2. a recorded UTC timestamp
3. a creator claim record
4. a unique certificate ID
5. a downloadable PDF certificate
6. a public verification page

POSITIONING:
OriginLock is not a copyright registration replacement.
OriginLock is not legal advice.
OriginLock is a proof-of-existence, timestamped creator evidence, and verification certificate system for creators who need documentation before sharing work with collaborators, clients, labels, or partners.

PRIMARY WEDGE:
Pitch protection and collaboration proof for independent creators.

SUPPORTED FILE TYPES FOR MVP:

DATABASE REQUIREMENTS:
Use Prisma with PostgreSQL.

Create these Prisma models:
- User
- CreatorProfile
- Project
- FileRecord
- TimestampRecord
- Certificate
- VerificationEvent
- AuditLog

Use this schema structure exactly unless a small technical adjustment is required for implementation:

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

enum PrivacyMode {
  private
  shareable_certificate
  public_verification
}

enum CertificateStatus {
  valid
  revoked
  superseded
}

enum VerificationResult {
  match
  mismatch
  not_found
  invalid
}

model User {
  id              String          @id @default(uuid())
  email           String          @unique
  passwordHash    String?
  emailVerified   Boolean         @default(false)
  createdAt       DateTime        @default(now())
  updatedAt       DateTime        @updatedAt

  creatorProfile  CreatorProfile?
  projects        Project[]
  files           FileRecord[]
  auditLogs       AuditLog[]
}

model CreatorProfile {
  id             String      @id @default(uuid())
  userId         String      @unique
  displayName    String
  creatorHandle  String      @unique
  legalName      String?
  pseudonym      String?
  bio            String?
  websiteUrl     String?
  claimStatement String?
  createdAt      DateTime    @default(now())
  updatedAt      DateTime    @updatedAt

  user           User        @relation(fields: [userId], references: [id], onDelete: Cascade)
  files          FileRecord[]
}

model Project {
  id           String       @id @default(uuid())
  userId       String
  title        String
  description  String?
  createdAt    DateTime     @default(now())
  updatedAt    DateTime     @updatedAt

  user         User         @relation(fields: [userId], references: [id], onDelete: Cascade)
  files        FileRecord[]
}

model FileRecord {
  id               String         @id @default(uuid())
  userId           String
  projectId        String?
  creatorProfileId String
  originalFilename String
  storedFilename   String?
  fileType         String
  mimeType         String
  fileSizeBytes    BigInt
  sha256Hash       String         @db.Char(64)
  storagePath      String?
  title            String
  notes            String?
  privacyMode      PrivacyMode    @default(private)
  createdAt        DateTime       @default(now())
  updatedAt        DateTime       @updatedAt

  user             User           @relation(fields: [userId], references: [id], onDelete: Cascade)
  project          Project?       @relation(fields: [projectId], references: [id], onDelete: SetNull)
  creatorProfile   CreatorProfile @relation(fields: [creatorProfileId], references: [id], onDelete: Cascade)
  timestamp        TimestampRecord?
  certificate      Certificate?
}

model TimestampRecord {
  id                String      @id @default(uuid())
  fileId            String      @unique
  recordedAtUtc     DateTime
  providerName      String?
  providerReference String?
  providerPayload   Json?
  createdAt         DateTime    @default(now())

  file              FileRecord  @relation(fields: [fileId], references: [id], onDelete: Cascade)
}

model Certificate {
  id                 String              @id @default(uuid())
  fileId             String              @unique
  certificateId      String              @unique
  status             CertificateStatus   @default(valid)
  verificationToken  String              @unique
  publicUrl          String
  pdfPath            String?
  qrCodePath         String?
  createdAt          DateTime            @default(now())
  updatedAt          DateTime            @updatedAt

  file               FileRecord          @relation(fields: [fileId], references: [id], onDelete: Cascade)
  verificationEvents VerificationEvent[]
}

model VerificationEvent {
  id              String              @id @default(uuid())
  certificateDbId String
  eventType       String
  ipAddress       String?
  userAgent       String?
  submittedHash   String?             @db.Char(64)
  result          VerificationResult
  createdAt       DateTime            @default(now())

  certificate     Certificate         @relation(fields: [certificateDbId], references: [id], onDelete: Cascade)
}

model AuditLog {
  id         String    @id @default(uuid())
  userId     String?
  entityType String
  entityId   String?
  action     String
  metadata   Json?
  createdAt  DateTime  @default(now())

  user       User?     @relation(fields: [userId], references: [id], onDelete: SetNull)
}
